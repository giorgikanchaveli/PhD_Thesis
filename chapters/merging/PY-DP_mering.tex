
Now we provide a result about merging between Pitman-Yor and Dirichlet process. Before, we we recall the posterior derived from Pitman-Yor process. Strategy for staritng merging rate is to first obtain the scaled version of the posterior distributions from each of the completely random measure, and then study the Wasserstein over adapted extended wasserstein discrepancy.

Let $\tilde{\mu}$ be Cox $\mathrm{CRM}(\tilde{\nu})$ according to the definition \ref{definition:Pitman_Yor_2}. The random L\'evy intesnity has the form
\[
\mathrm{d}\tilde{\nu}_\xi(s,x) = \frac{\xi \sigma}{\Gamma(1-\sigma)}\frac{e^{-s}}{s^{1+s}}\mathrm{d}s\mathrm{d}P_0(x),
\]
where $\xi \sim \Gamma\left(\frac{\theta}{\sigma}, 1 \right)$.

The posterior distribution for $\tilde{\mu}$ according to the Bayesian model \ref{model:bayesian_model_crm} has the following form (see Page 114 in \cite{hjort2010bayesian}:


Let $\tilde{\mu}$ be Cox CRM associated to the Pitman-Yor process. 

Given the observations $x_1,\dots, x_n$  from the model $\ref{model:bayesian_model_crm}$, the posterior version of the $\tilde{\mu}$ is a Cox CRM, denoted by $\mustar$ is characterized by  
\[
\mustar\ | U \overset{d}{=} \tilde{\mu} + \sum_{i = 1}^k J_i^U \delta_{x_i^*},
\]
where
\begin{itemize}
    \item $\tilde{\mu}$ is generalized gamma CRM with L\'evy intensity
    \[
    \mathrm{d}\tilde{\nu}(s,x) = \frac{ \sigma}{\Gamma(1 - \sigma)} \frac{e^{-Us}}{s^{1+\sigma}} \mathrm{d}s \mathrm{d}P_0(x).
    \]
    Note that we have ommitted the subscribt $U$ for the brevity of notation.
    \item $J_i^U \sim \mathrm{Gamma}(n_i - \sigma, u)$
    \item $f_U(u) =\frac{\sigma}{\Gamma\left(k+\frac{\theta}{\sigma}\right)} u^{\theta + k\sigma - 1}e^{-u^\sigma}\mathbf{1}_{(0,\infty)}(u)$
\end{itemize}
and $\left(x_1^*,\dots, x_k^*\right)$ are unique atoms with counts $n_1,\dots, n_k.$

Note that both $k$ and $U$ depend on n and they are seuqnces, however before starting merging rates, as we do not need to consider the asymptotic cases we drop the subcript for n for the brevity of notation.
\begin{proposition} 
    \label{proposition:posterior_pitmanyor}
    Let $\tilde{\mu}$ be Cox CRM associated to Pitman-Yor process with parameters $(\sigma, \theta, P_0)$, and $\mustar$ be its posterior. Then, the scaled version of $\mustar$ has the random L\'evy intensity 
    \[
    \mathrm{d}\tilde{v}^*_{S}(s,x) =  \mathrm{d}\rho_x^*(s)\mathrm{d}P^*_0(x),
    \]
    where
    \begin{align*}
        & P^*_0 = \frac{ \sigma U^\sigma}{C}P_0 + \frac{1}{C}\sum_{i = 1}^k (n_i - \sigma) \delta_{x_i^*},\\
        & \mathrm{d}\rho^*_x(s) = \frac{C^{1-\sigma}}{\Gamma(1-\sigma)}\frac{e^{-Cs}}{s^{1+\sigma}}\mathbf{1}_{\X \backslash\{x_1^*,\dots, x_k^*\}}(x)\mathrm{d}s + C\frac{e^{-Cs}}{s}\mathbf{1}_{\{x_1^*,\dots, x_k^*\}}(x)\mathrm{d}s, \\
        & C = C(\sigma, n, k, U) =  \sigma U^\sigma + n - k\sigma,
    \end{align*}
    and $U$ is defined as above.
\end{proposition}
\begin{proof}
    

Since CRM $\mustar |U$ is sum of CRMs, let us compute its L\'evy intensity. 

By iteratively applying lemma SM$4$ in \cite{catalano2023merging}, for general CRM $\tilde{\mu}$ with L\'evy intensity $\nu$, independent and infinitely divisible $J_1,\dots, J_k$ with L\'evy measures $\pi_i,$ and fixed atoms $y_1,\dots, y_k,$ the L\'evy intensity of $\tilde{\mu} + \sum_{i = 1}^k J_i \delta_{y_i}$ is 
\[
\mathrm{d}v(s,x) + \sum_{i = 1}^k \mathrm{d}\left( \pi_i \otimes \delta_{y_i} \right)(s,x), 
\]
where $v$ is L\'evy intensity of $\tilde{\mu}.$

For fixed $u > 0$, L\'evy measure associated to $\mathrm{Gamma}\left(n_i - \sigma, u\right)$
\[
\pi_i(s) = \frac{1}{s}(n_i - \sigma) e^{-us}\mathbf{1}_{(0,\infty)}(s)\mathrm{d}s.
\]
Then, for $(s,x) \in (0,\infty)\times\X,$ 
\begin{align*}
    \mathrm{d}v(s,x) + &\sum_{i = 1}^k \mathrm{d}\left( \pi_i \otimes \delta_{y_i} \right)(s,x)  \\
    & = \frac{ \sigma}{\Gamma(1-\sigma)}\frac{e^{-Us}}{s^{1+\sigma}}\mathrm{d}s \mathrm{d}P_0(x) + \sum_{i=1}^k \frac{1}{s}(n_i - \sigma)e^{-Us}\mathrm{d}s\mathrm{d}\delta_{x_i^*}(x) \\
    & = \frac{\sigma}{\Gamma(1-\sigma)}\frac{e^{-Us}}{s^{1+\sigma}}\mathrm{d}s \mathrm{d}P_0(x)  +\frac{e^{-Us}}{s}\mathrm{d}s\sum_{i = 1}^k (n_i - \sigma)\mathrm{d}\delta_{x_{^*}}(x)=: \mathrm{d}\tilde{v}^*(s,x).
\end{align*}
So, $\mustar$ is COx CRM with random L\'evy intensity $\tilde{v}^*.$ Denote by 
\begin{align}
    & \mathrm{d}\rho^1(s) = \frac{ \sigma}{\Gamma(1-\sigma)}\frac{e^{-Us}}{s^{1+\sigma}}\mathrm{d}s,\\
    & \mathrm{d}\rho^2(s) = \frac{e^{-Us}}{s}\mathrm{d}s.
\end{align}
then, 
\[
\mathrm{d}\tilde{v}^*(s,x) = \mathrm{d}\rho^1(s)\mathrm{d}P_0(x) + \mathrm{d}\rho^2(s)\sum_{i = 1}^k (n_i - \sigma)\mathrm{d}\delta_{x_{^*}}(x),
\]
next step is to obtain the scaled version of $\mustar$. Recall that due to definition \ref{definition:scaled_cox_crm} we obtain the scaled version of the latent CRM for fixed $U.$

As mentioned in subsectoin \ref{subsection:normalization_identifiability} if $v$ is L\'evy intensity of some CRM $\tilde{\mu}$, then its scaled CRM is 
\[
\mathrm{d}v_S(s,x) = \E \left(\tilde{\mu}(\X)\right) \rho_x\left( \E \left(\tilde{\mu}(\X)\right)s\right)\mathrm{d}P_0(x).
\]
due to Lemma SM3 in \cite{catalano2023merging}. By applying exactly same proof but for L\'evy inteinsity $\mathrm{d}\tilde{v}^*(s,x) = \rho^1(s)\mathrm{d}P^1_0(x) + \rho^2(s)\mathrm{d}P_0^2(x),$ we get
\[
\mathrm{d}v_S(s,x) =  \E \left(\tilde{\mu}(\X)\right) \rho^1_x\left(\E \left(\tilde{\mu}(\X)\right)s\right)\mathrm{d}P^1_0(x) + 
                       \E \left(\tilde{\mu}(\X)\right) \rho^1_x\left( \E \left(\tilde{\mu}(\X)\right)s\right)\mathrm{d}P^2_0(x).
\]
Therefore, let us first obtain $\E \left(\tilde{\mu}(\X)\right | U).$
By Cambpell's theorem,
\[
\E \left(\tilde{\mu}(\X)\right | U)  = \int_{(0,\infty)\times\X}s\mathrm{d}\tilde{v}^*(s,x).
\]
So,
\begin{align*}
    \E \left(\tilde{\mu}(\X)\right | U) & =\int_{(0,\infty)\times\X}s\frac{ \sigma}{\Gamma(1-\sigma)}\frac{e^{-Us}}{s^{1+\sigma}}\mathrm{d}s\mathrm{d}P_0(x) 
    +\int_{(0,\infty)\times\X}s\frac{e^{-Us}}{s}\mathrm{d}s\sum_{i = 1}^k(n_i - \sigma)\mathrm{d}\delta_{x_i^*}(x) \\
    & = \frac{ \sigma}{\Gamma(1-\sigma)}\int_0^\infty s^{-\sigma} e^{-Us}\mathrm{d}s + (n-k\sigma)\int_0^\infty e^{-Us}\mathrm{d}s = \\
    & = \frac{ \sigma}{\Gamma(1-\sigma)} \frac{\Gamma(1-\sigma)}{U^{1-\sigma}} + \frac{n - k\sigma}{U} \\
    & = \frac{ \sigma U^\sigma}{U} + \frac{n - k\sigma}{U} = \frac{C}{U},
\end{align*}
where
$C = C( \sigma, n, k, U) =  \sigma U^\sigma + n - k\sigma.$

Then by pluging it in the formula for scaled  L\'evy intensity, we get that the scaled L\'evy intensity of latent CRM is
\begin{align*}
    \mathrm{d}\tilde{v}^*_{S}(s,x) &= \frac{C}{U}\rho_1\left(\frac{C}{U}s \right)\mathrm{d}P_0(x) + 
    \frac{C}{U}\rho_2\left(\frac{C}{U}s \right)\sum_{i = 1}^k(n_i-\sigma)\mathrm{d}\delta_{x_i^*}(x) \\
    & = \frac{C}{U}\frac{ \sigma}{\Gamma(1-\sigma)}\frac{e^{-Cs}}{C^{1+\sigma}s^{1+\sigma}}U^{1+\sigma}\mathrm{d}s\mathrm{d}P_0(x) 
    + \frac{e^{-Cs}}{s}\mathrm{d}s \sum_{i = 1}^k(n_i-\sigma)\mathrm{d}\delta_{x_i^*}(x) \\ 
    & = \frac{ \sigma}{\Gamma(1-\sigma)}\frac{e^{-Cs}}{s^{1+\sigma}}\left(\frac{U}{C} \right)^\sigma\mathrm{d}s \mathrm{d}P_0(x) \\
    & + \frac{e^{-Cs}}{s}\mathrm{d}s\sum_{i  =1}^k (n_i - \sigma) \mathrm{d}\delta_{x_i^*}(x).
\end{align*}
The problem now is that it is not written in the canonical form. We want to get product between density for atoms and transition kernel for jumps. We will now write it in the canonical form.

Note that 
\[
\mathcal{M}_1\left(\tilde{v}^*_{S}\right) = \int_{(0, \infty)\times \X }s \mathrm{d}\tilde{v}^*_{S}(s,x) = 1,
\]
as it is L\'evy intensity for scaled CRM. Then, the density $s\mathrm{d}\tilde{v}^*_{S}$ can be disintegrated as 
\[
s\mathrm{d}\tilde{v}^*_{S} = \mathrm{d} \varrho_x(s)\mathrm{d}Q(x),
\]
where $Q$ is second marginal of $s\mathrm{d}\tilde{v}^*_{S}$ and $\varrho_x$ is the probability transition kernel.

Note that,
\[
Q(\X) = \int_{(0,\infty)\times \X}s\mathrm{d}\tilde{v}^*_{S}(s,x) = 1.
\]
This implies that $Q$ is a probabilty measure. Then, we get
\[
\mathrm{d}\tilde{v}^*_{S}(s,x) = \frac{1}{s}\mathrm{d}\varrho_x(s)\mathrm{d}Q(x).
\]
To find the terms in the decomposition, we first find the form of $Q(\cdot).$

For $A \in \mathcal{B}(\X),$
\begin{align*}
    Q(A) &= \int_{(0,\infty)\times A}s\mathrm{d}\tilde{v}^*_{S}(s,x) \\
    & = \frac{ \sigma}{\Gamma(1-\sigma)}\left( \frac{U}{C} \right)^\sigma P_0(A) \int_0^\infty s^{-\sigma}e^{-Cs}\mathrm{d}s 
    + \sum_{i = 1}^k (n_i - \sigma) \delta_{x_i^*}(A) \int_0^\infty e^{-Cs}\mathrm{d}s \\ 
    &= \frac{ \sigma}{\Gamma(1-\sigma)}\left( \frac{U}{C}\right)^\sigma P_0(A) \frac{\Gamma(1-\sigma)}{C^{1-\sigma}} 
    + \frac{1}{C} \sum_{i = 1}^k(n_i - \sigma)\delta_{x_i^*}(A) \\
    & = \frac{\sigma U^\sigma}{C} P_0(A) + \frac{1}{C}\sum_{i = 1}^k (n_i - \sigma) \delta_{x_i^*}(A).
\end{align*}
We denote by 
\[
P_0^*(\cdot) = \frac{\sigma U^\sigma}{C} P_0(\cdot) + \frac{1}{C}\sum_{i = 1}^k (n_i - \sigma) \delta_{x_i^*}(\cdot).
\]
Now $\varrho_x$ should satisfy
\[
\frac{ \sigma}{\Gamma(1-\sigma)}\frac{e^{-Cs}}{s^{\sigma+1}}\left( \frac{U}{C} \right)^\sigma\mathrm{d}s \mathrm{d}P_0(x) + 
\frac{e^{-Cs}}{s}\mathrm{d}s \sum_{i = 1}^k(n_i - \sigma) \mathrm{d}\delta_{x_i^*}(x) = \frac{1}{s}\varrho_x(s)\mathrm{d}P^*_0(x).
\]
As $P_0$ is non-atomic, we obtain that 
\[
\rho_x(s) = \frac{C^{1-\sigma}}{\Gamma(1-\sigma)}s^{-\sigma}e^{-Cs}\mathbf{1}_{\X \backslash\{x_1^*,\dots, x_k^*\}}(x)\mathrm{d}s + 
Ce^{-Cs}\mathbf{1}_{\{x_1^*,\dots, x_k^*\}}(x)\mathrm{d}s.
\]
Then, the conclusion is obtained by defining 
\[
\rho_x^*(s) = \frac{1}{s}\varrho_x(s) = \frac{C^{1-\sigma}}{\Gamma(1-\sigma)}\frac{e^{-Cs}}{s^{1+\sigma}}\mathbf{1}_{\X \backslash\{x_1^*,\dots, x_k^*\}}(x)\mathrm{d}s + C\frac{e^{-Cs}}{s}\mathbf{1}_{\{x_1^*,\dots, x_k^*\}}(x)\mathrm{d}s.
\]
\end{proof}


Now we are ready to study the merging between Pitman-Yor and Dirichlet process. 
\begin{proposition}\label{proposition:distance_pitmanyor_gamma}
    Let $\tilde{\mu}^1$ be Pitman-Yor CRM with parameters $(\sigma, \theta,  P_0)$ and $\tilde{\mu}^2$  be gamma CRM with parameters $(\alpha, P_0)$. Then
    \[
    \wad\left( \law\left(\tilde{v}^{*,1}_{S}\right),  \law\left(\tilde{v}^{*,2}_{S}\right) \right)  = \E_U\left(\mathcal{A}^U + J^U_1 + J^2_U \right),
    \]
    where
    \begin{align*}
        & \mathcal{A}^U = \W_{\mathrm{d}_{\X,t}} \left( \frac{ \sigma U^\sigma}{C}P_0 + \frac{1}{C}\sum_{i = 1}^{k_n}(n_i - \sigma)\delta_{x_i^*}, \frac{\alpha}{\alpha + n}P_0 + \frac{1}{\alpha + n}\sum_{i = 1}^n \delta_{x_i}\right), \\
        & J_1^U = \frac{ \sigma U^\sigma}{C}\W_*\left( \frac{C^{1-\sigma}}{\Gamma(1-\sigma)}\frac{e^{-Cs}}{s^{1+\sigma}}\mathrm{d}s, (\alpha+n)\frac{e^{-(\alpha+n)s}}{s}\mathrm{d}s\right), \\
        & J_2^U = \frac{n - k\sigma}{C}\W_*\left( \frac{Ce^{-Cs}}{s}\mathrm{d}s, (\alpha + n)\frac{e^{-(\alpha+n)s}}{s}\mathrm{d}s\right),
    \end{align*}
    with $U $ and $C $ are defined as in proposition \ref{proposition:posterior_pitmanyor}
\end{proposition}
\begin{proof}
    Denote by $\tilde{v}^{*,2}_{S}$ L\'evy intensity for the scaled gamma posterior. Its derivation can be found in proof of lemma 12 in \cite{catalano2023merging}, it has the follwoing form:
    \[
    \mathrm{d}\tilde{v}^{*,2}_{S}(s,x) = (\alpha + n)\frac{e^{(\alpha + n)s}}{s}\left( \frac{\alpha}{\alpha + n} \mathrm{d}P_0(x) + \frac{1}{\alpha + n}\sum_{i = 1}^n \delta_{x_i}(x) \right)
    \]
    Note that the scaled gamma posterior is no longer Cox CRM. Then, as it is homogeneous, due to Remark \ref{remark:adapted_extwas_homogeneous_crms}
    Therefore,
    \begin{align*}
    \wad\left( \law\left(\tilde{v}^{*,1}_{S}\right),  \law\left(\tilde{v}^{*,2}_{S}\right) \right) &= \E_{U \sim f_U}\left(\ad(\tilde{v}^{*,1}_{S}, v^{*,2}_S \right) \\
    &= \E_{U \sim f_U}  \W_{\mathrm{d}_{\X,t}} \left(P_0^{*,1}, P_0^{*,2} \right) + \E_{U \sim f_U}  \E_{X \sim P_0^{*,1}}\W_*\left(\rho_X^{*,1} , \rho^{*,2}\right),
    \end{align*}
    where 
    \begin{itemize}
        \item $\rho_x^{*,1}(s) = \frac{C^{1-\sigma}}{\Gamma(1-\sigma)}\frac{e^{-Cs}}{s^{1+\sigma}}\mathbf{1}_{\X \backslash\{x_1^*,\dots, x_k^*\}}(x)\mathrm{d}s + C\frac{e^{-Cs}}{s}\mathbf{1}_{\{x_1^*,\dots, x_k^*\}}(x)\mathrm{d}s.$
        \item $\rho^{*,2}(s)(\alpha + n)\frac{e^{(\alpha + n)s}}{s}$.
        \item $P_0^{*,1}(\cdot) = \frac{\sigma U^\sigma}{C} P_0(\cdot) + \frac{1}{C}\sum_{i = 1}^k (n_i - \sigma) \delta_{x_i^*}(\cdot).$
        \item $P_0^{*,2}=\frac{\alpha}{\alpha + n} \mathrm{d}P_0(x) + \frac{1}{\alpha + n}\sum_{i = 1}^n \delta_{x_i}(x)$
    \end{itemize}
    The first term in the sum is 
    \[
    \E_{U \sim f_U}  \W_{\mathrm{d}_{\X,t}} \left(\frac{ \sigma U^\sigma}{C}P_0 + \frac{1}{C}\sum_{i = 1}^{k_n}(n_i - \sigma) \delta_{x_i^*},
                         \frac{\alpha}{\alpha + n}P_0 + \frac{1}{\alpha + n}\sum_{i = 1}^n\delta_{x_i}\right) = \E_{U \sim f_U} \mathcal{A}^U.
    \]
    The second term is 
    \begin{align*}
        \E_{U \sim f_U}  \E_{X \sim P_0^{*,1}}\W_*\left(\rho_X^{*,1} , \rho^{*,2}\right) & = \E_{U \sim f_U} \frac{ \sigma U^\sigma}{C}\E_{X \sim P_0}\W_*\left(\rho_X^{*,1} , \rho^{*,2} \right) 
        + \E_{U \sim f_U}\frac{1}{C}\sum_{i = 1}^{k_n} (n_i - \sigma)\W_*\left(\rho_{x_i}^{*,1} , \rho^{*,2} \right) \\
        & =  \E_{U \sim f_U} \frac{ \sigma U^\sigma}{C}\W_*\left(\frac{C^{1-\sigma}}{\Gamma(1-\sigma)}\frac{e^{-Cs}}{s^{1+\sigma}}\mathrm{d}s,
                (\alpha + n)\frac{e^{-(\alpha + n)s}}{s}\mathrm{d}s\right) \\
        & +  \E_{U \sim f_U} \frac{n-k\sigma}{C}\W_*\left( \frac{Ce^{-Cs}}{s}\mathrm{d}s, (\alpha + n)\frac{e^{-(\alpha + n)s}}{s}\mathrm{d}s\right) \\
        & = \E_{U \sim f_U} J_1^U + \E_{U \sim f_U} J_2^U.
    \end{align*}
\end{proof}

\begin{theorem}
    Let $\tilde{\mu}^1, \tilde{\mu}^2$ as in Proposition \ref{proposition:distance_pitmanyor_gamma} and let $(x_n)_{n\geq 1}$ be any sequence in $\R.$
    Then as $n \to \infty,$
    \[
    \wad \left( \law\left(\tilde{v}^{*,1}_{S}\right),  \law\left(\tilde{v}^{*,2}_{S}\right) \right) \asymp \frac{k_n}{n.}
    \]
    In particular, $\wbl \left( \law\left(\tilde{v}^{*,1}_{S}\right),  \law\left(\tilde{v}^{*,2}_{S}\right) \right) \lesssim \frac{k_n}{n}.$
\end{theorem}
\begin{proof}
    Firstly, note that bound for the Wasserstein over Bounded Lipschitz folows from Theorem $6$ of \cite{catalano2023merging}.

    Now we use the subsscript for $n$ for $k_n$ and random variables $U_n, C_n$. Recall that $U_n$ has density functnio $f_U(u) =\frac{\sigma}{\Gamma\left(k+\frac{\theta}{\sigma}\right)} u^{\theta + k\sigma - 1}e^{-u^\sigma}\mathbf{1}_{(0,\infty)}(u)$. By change of variable $y = u^\sigma$, we obtain that the random variable $U_n^\sigma \sim \mathrm{Gamma}\left(k_n + \frac{\theta}{\sigma} \right).$
    
    Following the result from Proposition \ref{proposition:distance_pitmanyor_gamma}, we study the behaviour of each term in $\E_U\left(J^{U_n}_1 + J_2^{U_n} + \mathcal{A}^{U_n}\right)$.
    \begin{itemize}
        \item First term: $\E_{U_n} J^{U_n}_1$
    \end{itemize}
    Recall that,
    \[
    J_1^U = \frac{ \sigma U_n^\sigma}{C_n}\W_*\left( \frac{C_n^{1-\sigma}}{\Gamma(1-\sigma)}\frac{e^{-C_ns}}{s^{1+\sigma}}\mathrm{d}s, (\alpha+n)\frac{e^{-(\alpha+n)s}}{s}\mathrm{d}s\right) =  \frac{ \sigma U_n^\sigma}{C_n} W_n,
    \]
    where $C_n = \sigma U_n^\sigma + n - k_n\sigma$ and 
    \[
    W_n = \W_*\left( \frac{C_n^{1-\sigma}}{\Gamma(1-\sigma)}\frac{e^{-C_ns}}{s^{1+\sigma}}\mathrm{d}s, (\alpha+n)\frac{e^{-(\alpha+n)s}}{s} \mathrm{d}s\right)
    \]
    

    Let us first obtain the upper bound for the expected value.

    Note that the extended wasserstein distance can be upper bounded. Indeed
    \begin{align*}
        W_n &= 
        \int_0^\infty \left| \int_s^\infty \frac{C_n^{1-\sigma}}{\Gamma(1-\sigma)}\frac{e^{-C_nt}}{t^{1+\sigma}}\mathrm{d}t - 
        \int_s^\infty (\alpha+n)\frac{e^{-(\alpha+n)t}}{t} \mathrm{d}t \right| \mathrm{ds}\\ 
        & \leq \int_0^\infty  \int_s^\infty \frac{C_n^{1-\sigma}}{\Gamma(1-\sigma)}\frac{e^{-C_nt}}{t^{1+\sigma}}\mathrm{d}t\mathrm{d}s + 
        \int_0^\infty\int_s^\infty (\alpha+n)\frac{e^{-(\alpha+n)t}}{t} \mathrm{d}t\mathrm{d}s \\
        & \leq \int_0^\infty s \frac{C_n^{1-\sigma}}{\Gamma(1-\sigma)}\frac{e^{-C_ns}}{s^{1+\sigma}}\mathrm{d}s + 
        \int_0^\infty s  (\alpha+n)\frac{e^{-(\alpha+n)s}}{s} \mathrm{d}s,
    \end{align*}
    where last inequality follows from Fubini's theorem. As the last two integrals equal to $1$, we get that $W_n \leq 2.$
    
    
    Now note also that 
    \[
    \frac{\sigma U_n^\sigma}{C_n} = \frac{\sigma U_n^\sigma}{\sigma U_n^\sigma + n - k_n\sigma} \leq \frac{\sigma}{1-\sigma}\frac{U_n^\sigma}{n}.
    \]
    Taking the expectation we get,
    \[
    \E \frac{\sigma U_n^\sigma}{C_n} \leq \frac{\sigma}{1-\sigma}\frac{k_n + \frac{\theta}{\sigma}}{n}.
    \]
    So we get that
    \[
    \E J_1^{U_n} \leq 2\frac{\sigma}{1-\sigma}\frac{k_n + \frac{\theta}{\sigma}}{n},
    \]
    implying that $\E J_1^{U_n}\lesssim \frac{k_n}{n}.$

    Now let us obtain the lower bound. For that we distinguish the two caes:
    \[
    \sup k_n < \infty \quad \text{and} \quad \sup k_n = \infty.
    \]
    \begin{itemize}
        \item Case: $\sup k_n < \infty$
    \end{itemize}
    As $k_n$ is nondecreasing sequence of integers, there exists $N \geq 1,$ so that for all $n \geq N $ we have $k_n = k$ for $k = \sup k_n.$ This implies that for all $n \geq N$,
    \[
    U_n^\sigma \sim \mathrm{Gamma}\left(k +\frac{\theta}{\sigma},1\right).
    \]
    Let $p \in (0,1)$ and $t_1<t_2$ such that for all $n \geq N,$
    \[
    \mathbb{P}\left(t_1 < U_n^\sigma < t_2\right) = p.
    \]
    Denote by $A_n = \left\{ t_1 < U_n^\sigma < t_2 \right\}$. Then on $A_n,$
    \[
    \frac{\sigma U_n^\sigma}{C_n} = \frac{\sigma U_n^\sigma}{\sigma U_n^\sigma + n - k_n\sigma} \geq \frac{t_1 \sigma}{t_2 \sigma + n(1-\sigma)}.
    \]
    Now we lower bound the $W_n$ on set $A_n$ by some positive value.

    \begin{align*}
        \W_*\Bigg(\frac{C_n^{1-\sigma}}{\Gamma(1-\sigma)}&\frac{e^{-C_ns}}{s^{1+\sigma}}\mathrm{d}s, 
                  (\alpha + n)\frac{e^{-(\alpha + n)s}}{s}\mathrm{d}s\Bigg)  \\
        & =\int_0^\infty \left| \int_s^\infty \frac{C_n^{1-\sigma}}{\Gamma(1-\sigma)}\frac{e^{-C_nt}}{t^{1+\sigma}}\mathrm{d}t - \int_s^\infty (\alpha + n)\frac{e^{-(\alpha + n)t}}{t}\mathrm{d}t\right|\mathrm{d}s \\ 
        & = \int_0^\infty \left|\int_{C_ns}^\infty \frac{C_n^{1-\sigma}}{\Gamma(1-\sigma)}\frac{e^{-y}}{y^{1+\sigma}}C_n^{1+\sigma} \frac{1}{C_n}\mathrm{d}y - 
        \int_{(\alpha+n)s}^\infty (\alpha + n) \frac{e^{-y}}{y}\mathrm{d}y\right| \\
        & = \int_0^\infty \left| \frac{C_n}{\Gamma(1-\sigma)}\Gamma\left(-\sigma, C_ns\right) - (\alpha + n)\Gamma(0, (\alpha + n)S)\right|\mathrm{d}s,
    \end{align*}
    where $\Gamma(z,t) = \int_t^\infty x^{z-1}e^{-x}\mathrm{d}x$ denotes the incomplete gamma function.

    Using the change of variable $t = C_ns$, the last expression is further reduced to 
    \[
    \int_0^\infty \left| \frac{\Gamma(-\sigma, t)}{\Gamma(1-\sigma)}- \frac{\alpha + n}{C_n}\Gamma\left( 0, \frac{\alpha + n}{C_n}t\right)\right|\mathrm{d}t.
    \]
    Note that on $A_n$
    \[
    \frac{n}{t_2\sigma + n}\leq \frac{\alpha + n}{\sigma U_n^\sigma + n - k_n \sigma} \leq \frac{\alpha + n}{t_1 \sigma + n(1-\sigma)}
    \]
    implying taht there exists $N' \geq N$ such that for all $n \geq N'$, on $A_n$, we have that $\frac{\alpha + n}{C_n} \in \left[\frac{1}{2},2\right].$
    Define the functoin $F : (0,\infty) \to \R$,
    \[
    F(r) = \int_0^\infty \left| \frac{\Gamma(-\sigma, t)}{\Gamma(1-\sigma)}- r\Gamma\left( 0, rt\right)\right|\mathrm{d}t.
    \]
    Then the function $F$ is continuous. We want to show that $F$ is bounded below by some positive number on $\left[\frac{1}{2},2\right]$ which will allow us to deduce that $W_n$ is bounded below on $A_n.$

    Suppose that $F(r) = 0$ for some $r \in  \left(\frac{1}{2}, 2\right).$ Then as the function we integrate is continouous we get that  for every $t \in (0,\infty)$,
    \[
    \left| \frac{\Gamma(-\sigma, t)}{\Gamma(1-\sigma)}- r\Gamma\left( 0, rt\right)\right| = 0.
    \]
    So we get 
    \[
    \frac{\Gamma(-\sigma, t)}{1-\sigma)} = r \Gamma(0, rt).
    \]
    By differentiating with respect to $t$, we get that for all $t >0$
    \[
    -\frac{t^{-(\sigma-1)}e^{-t}}{\Gamma(1-\sigma)} = -r^2 (tr)^{-1}e^{-tr},
    \]
    that is
    \[
    t^{-\sigma}e^{t(r-1)} = r\Gamma(1-\sigma).
    \]
    As left hand side is constant but right hand side not, we get the contradiction. Therefore, $F(r) > 0$ for all $r \in \left[\frac{1}{2},2\right].$ Then $F$ attains minimum on that compact interval. This means that on $A_n$, as $\frac{\alpha + n}{C_n} \in \left[\frac{1}{2},2\right],$ we have that $W_n > w$ for some $w > 0.$    

    Finally we have,
    \[
    \E \frac{\sigma U_n^\sigma W_n}{C_n} \geq \E \frac{\sigma U_n^\sigma W_n}{C_n}\mathbf{1}_{A_n} \geq \frac{t_1 \sigma}{t_2 \sigma + n(1-\sigma)}w p 
    \]
    for  $n \geq N'.$ Together with upper bound, We deduce that $\E J_1^{U_n} \asymp \frac{1}{n} $

    \begin{itemize}
        \item Case: $\sup k_n = \infty.$
    \end{itemize}
    For that our strategy is to show that the random variable $\frac{n}{k_n}J_1^{U_n}$ converges in probability to something positive and is uniformly integrable. This will allow us to obtain the L-1 convergence and bound the expecation asymptotically. 

    Let us start with showing the $L$-1 convergence of $\frac{U_n}{k_n}.$

    \[
    \E \left| \frac{U_n}{k_n} - 1\right|^2 = \E \left| \frac{U_n-k_n}{k_n}{}\right|^2 = \E \left| \frac{U_n-E U_n + \frac{\theta}{\sigma}}{k_n}{}\right|^2 \leq \frac{\var(U_n)}{k_n} + \left(\frac{\frac{\theta}{\sigma}}{k_n} \right)^2.
    \]
    As $\var(U_n) = k_n + \frac{\theta}{\sigma},$ we get 
    \[
    \lim_{n\to \infty}\E \left|\frac{U_n}{k_n} - 1 \right|^2 = 0.
    \]
    As $L$-2 convergence implies L 1, we get the desired result.

    Now we show that $\frac{n}{C_n}$ converges in probability to some positive value.
    \[
    \frac{n}{C_n} = \frac{n}{\sigma U_n^\sigma + n - k_n\sigma} = \frac{1}{\frac{\sigma U_n^\sigma}{n}+ 1 - \frac{k_n}{\sigma}}.
    \]
    Note that if $k_n \asymp n$, then by above, $\frac{U_n}{n} \overset{P}{\to} 1,$ otherwise it converges to $0$ in probability. In both of the cases we obtain that 
    \[
    \frac{n}{C_n} \overset{P}{\to} c
    \]
    for some $c > 0.$

    Now we show the convergence in probability of $W_n.$ As we showed above,
    \[
    W_n = \int_0^\infty \left| \frac{\Gamma(-\sigma, t)}{\Gamma(1-\sigma)}- \frac{\alpha + n}{C_n}\Gamma\left( 0, \frac{\alpha + n}{C_n}t\right)\right|\mathrm{d}t
    \]
    
    As we have showed above that $\frac{n}{C_n}$ converges in probability, so the last expression converges in probability to 
    \[
    \int_0^\infty \left| \frac{\Gamma(-\sigma, t)}{\Gamma(1-\sigma)} - c'\Gamma(0,c't) \right|\mathrm{d}t
    \]
    by continuous mapping theorem, where $c'$ is some positive constant.

    So we showed that $\frac{n}{k_n}J_1^{U_n}$ converges in probability to some positive constant $c^*.$ Now we show the uniform integrability.
    \[
    \frac{n}{k_n}\frac{\sigma U_n^\sigma}{C_n} W_n \leq 2\frac{n\sigma U_n^\sigma}{k_n(n-k_n\sigma)}\leq \frac{n \sigma U_n^\sigma}{k_n n(1-\sigma)} = \frac{\sigma}{1-\sigma}\frac{U_n^\sigma}{k_n}.
    \]
    Since the last term converges in $L-$1 , we get the uniform integrability.

    So we get that 
    \[
    \E \frac{n}{k_n}J_1^{U_n} \to c^*
    \] deducing that $\E j_1^{U_n} \asymp \frac{k_n}{n}.$

   
    \begin{itemize}
        \item Second term.
    \end{itemize}
    Note that 
    \[
    \frac{n-k_n\sigma}{C_n} = \frac{n-k_n\sigma}{ \sigma U_n^\sigma + n - k_n\sigma} \leq \frac{n-k_n\sigma}{n-k_n\sigma} = 1.
    \]
    So we study the behaviour of extended Wasserstein distance.
    \begin{align*}
    \W_*\Bigg( C_n\frac{e^{-C_ns}}{s}\mathrm{d}s, & (\alpha + n)\frac{e^{-(\alpha +n)s}}{s}\mathrm{d}s\Bigg) \\ 
    & = \int_0^\infty \left| \int_s^\infty C_n \frac{e^{-C_nt}}{t}\mathrm{d}t - \int_s^\infty (\alpha + n)\frac{e^{-(\alpha +n)t}}{t}\mathrm{d}t\right|\mathrm{d}s \\ 
    & = \int_0^\infty \left| \int_{C_ns}^\infty C_n \frac{e^{-y}}{y}\mathrm{d}y - \int_{(\alpha+n)s}^\infty (\alpha + n)\frac{e^{-y}}{y}\mathrm{d}y\right|\mathrm{d}s\\
    & = \int_0^\infty \left| C_n\Gamma\left(0,C_ns\right) - (\alpha + n)\Gamma(0, (\alpha + n)s) \right|\mathrm{d}s \\
    & = J\left(C_n-n,\alpha + n\right).
    \end{align*}
    where $J\left(\alpha_1, \alpha_2\right)$ is defined as
    \[
    J\left(\alpha_1, \alpha_2\right) = \int_0^\infty \left| (\alpha_1 + n)\Gamma(0,(\alpha_1 + n)s) - (\alpha_2+n)\Gamma(0,(\alpha_2 +n)s) \right|\mathrm{d}s.
    \] 
    By point $2$ of Proposition $14$ in \cite{catalano2023merging}, we have that for every $n \geq 1,$
    \[
    J\left(C_n - n, \alpha, n\right) \leq c \max\left\{ \log\left( \frac{C_n}{\alpha + n}\right), \log\left( \frac{\alpha + n}{C_n}\right)\right\},
    \]
    where $c  =\int_0^\infty \left| \Gamma(0, t) - e^{-t} \right|\mathrm{d}t.$

    Using the fact that $\log x \leq x - 1,$ we have that 
    \[
    J\left(C_n - n, \alpha, n\right) \leq c \max \left\{ \frac{C_n}{\alpha + n}-1, \frac{\alpha + n}{C_n} - 1\right\}
    \]
    We bound each term in the maximum seperately.

    \[
    \frac{C_n}{\alpha + n} - 1 = \frac{ \sigma U_n^\sigma - k_n \sigma - \alpha}{\alpha + n} \leq  \sigma \frac{U_n^\sigma}{\alpha + n}.
    \]
    then,
    \[
    \E \frac{ \sigma U_n^\sigma}{\alpha + n} = \frac{\sigma\left(k_n + \frac{\theta}{\sigma}\right)}{\alpha + n},
    \]
    we have that
    \[
    \E \left( \frac{C_n}{\alpha + n} - 1\right) \lesssim \frac{k_n}{n}.
    \]
    Now we bound the second term in the maximum.

    
    \[
    \frac{\alpha + n}{C_n} - 1 = \frac{\alpha -  \sigma U_n^\sigma + k_n \sigma}{n +  \sigma U_n^\sigma - k_n \sigma}\leq
    \frac{\alpha + k_n \sigma}{n(1- \sigma)}.
    \]
    So,
    \[
    E \left(\frac{\alpha + n}{C_n} - 1\right) \lesssim \frac{k_n}{n}.
    \]
    \begin{itemize}
        \item Third term.
    \end{itemize}
    Finally we bound the
    \[
    \mathcal{A}^{U_n} = \W_{\mathrm{d}_{\X,t}} \left( \frac{ \sigma U_n^\sigma}{C_n}P_0 + \frac{1}{C_n}\sum_{i = 1}^{k_n}(n_i - \sigma)\delta_{x_i^*}, \frac{\alpha}{\alpha + n}P_0 + \frac{1}{\alpha + n}\sum_{i = 1}^n \delta_{x_i}\right)
    \]
    The proof will use two convexity properties of the wasserstein distance. First property we use is that for all $\lambda \in (0,1)$ and $P^1,P^2,Q^1,Q^2 \in \probx$,
    \[
    \wtr\left( \lambda P^1 + (1-\lambda)Q^1, \lambda P^2 + (1-\lambda)Q^2 \right) \leq \lambda \wtr(P^1,P^2) + (1-\lambda)\wtr(Q^1,Q^2).
    \]
    Second property is that for all $\lambda \in (0,1)$ and $P,Q^1,Q^2 \in \probx$, we have 
    \[
    \wtr (\lambda  P + (1-\lambda)Q^1, \lambda P + (1-\lambda)Q^2) = (1-\lambda)\wtr(Q^1,Q^2).
    \]
    As in the Wasserstein distance, we have terms $P_0$ on both sides, we try to match the coefficient, as it will help us to cancel some terms. But problem is that we do not know which coefficient of $P_0$ is larger. So we have two cases:
    \[
    \frac{ \sigma U_n^\sigma}{C_n} > \frac{\alpha}{\alpha + n} \quad \text{and} \quad 
    \frac{ \sigma U_n^\sigma}{C_n} < \frac{\alpha}{\alpha + n}.
    \]
 
    We decompose the expectation on the sets where these conditions are satisfied. As we will see below, on both sets, the expectations will be asymptotically bounded by 
    \[
    \frac{k_n}{n}.
    \]

    Define the set 
    \[
    A_n = \left\{ \frac{ \sigma U_n^\sigma}{C_n} > \frac{\alpha}{\alpha + n}\right\}.
    \]
    So we have that 
    \[
    \E \mathcal{A}^U = \E \mathcal{A}^U\mathbf{1}_{A_n} + \E \mathcal{A}^U\mathbf{1}_{A^c_n}.
    \]
    On $A_n$, we decompose
    \[
    \frac{ \sigma U_n^\sigma}{C_n} P_0 = \frac{\alpha}{\alpha + n}P_0 + \left( \frac{ \sigma U_n^\sigma}{C_n} - \frac{\alpha}{\alpha + n} \right) P_0 = \frac{\alpha}{\alpha + n}P_0 + \lambda_1 P_0,
    \]
    where
    \[
    \lambda_1 = \frac{ \sigma U_n^\sigma}{C_n} - \frac{\alpha}{\alpha + n}.
    \]    
    Note that $\lambda_1 \in (0,1)$. Then
    \begin{align*}
        \wtr \Big(\frac{\alpha}{\alpha + n}&P_0 +  \frac{n}{\alpha + n} \left( \frac{\alpha + n}{n}\lambda_1 P_0 + \frac{\alpha + n}{n}\frac{n-k_n\sigma}{C_n}\sum_{i = 1}^{k_n}\frac{\left(n_i - \sigma\delta_{x_i^*}\right)}{n-k_n\sigma} \right),
       \frac{\alpha}{\alpha + n}P_0 + \frac{n}{\alpha + n}\sum_{i = 1}^n\frac{1}{n}\delta_{x_i}\Big) \\
        & \leq \frac{n}{\alpha + n}\wtr \left( \frac{\alpha + n}{n}\lambda_1 P_0 + \frac{\alpha + n}{n}\frac{n-k_n\sigma}{C_n}\sum_{i = 1}^{k_n}\frac{\left(n_i - \sigma \delta_{x_i^*}\right)}{n-k_n\sigma}, \sum_{i = 1}^n \frac{1}{n}\delta_{x_i}\right) \\ 
        & = \frac{n}{\alpha + n}\wtr \left( \frac{\alpha + n}{n}\lambda_1 P_0 + \frac{\alpha + n}{n}\frac{n-k_n\sigma}{C_n}\sum_{i = 1}^{k_n}\frac{\left(n_i - \sigma \delta_{x_i^*}\right)}{n-k_n\sigma}, \frac{\alpha + n}{n}\lambda_1\sum_{i = 1}^n\frac{1}{n}\delta_{x_i}  + \frac{\alpha + n}{n}\frac{n-k_n\sigma}{C_n}\sum_{i = 1}^n\frac{1}{n}\delta_{x_i}\right) \\
        & \leq \lambda_1 \wtr\left(P_0, \sum_{i = 1}^n \frac{1}{n}\delta_{x_i}\right) + \lambda_2\wtr \left( \sum_{i = 1}^{k_n} \frac{(n_i-\sigma)}{n-k_n\sigma}\delta_{x_i^*},\sum_{i = 1}^n\frac{1}{n}\delta_{x_i}\right),
    \end{align*}
    where 
    \[
    \lambda_2 = \frac{n - k_n \sigma}{C_n}.
    \]
    We have showed that
    \[
    \E \mathcal{A}^{U_n}\mathbf{1}_{A_n} \leq \E \left[ \mathbf{1}_{A_n}\lambda_1 \wtr\left(P_0, \sum_{i = 1}^n \frac{1}{n}\delta_{x_i}\right) +  \mathbf{1}_{A_n}\lambda_2\wtr \left( \sum_{i = 1}^{k_n} \frac{(n_i-\sigma)}{n-k_n\sigma}\delta_{x_i^*},\sum_{i = 1}^n\frac{1}{n}\delta_{x_i}\right) \right].
    \]
    Note that in the first term of the sum within expectation, since $d_{\X,t} \leq 2,$ we can bound Wasserstein distance by $2$, and $\lambda_1$ by $\frac{ \sigma U_n^\sigma}{C_n}$. Then,
    \[
    \E \left[ \mathbf{1}_{A_n}\lambda_1 \wtr\left(P_0, \sum_{i = 1}^n \frac{1}{n}\delta_{x_i}\right)\right]\leq \E \frac{ \sigma U_n^\sigma}{C_n} \leq \frac{\sigma}{1-\sigma}\frac{k_n + \frac{\theta}{\sigma}}{n},
    \]
    and so it is asymptotically bounded by $\frac{k_n}{n}$.
    
    Now note that, for bounding 
    \[
    \E \left[\mathbf{1}_{A_n}\lambda_2\wtr \left( \sum_{i = 1}^{k_n} \frac{(n_i-\sigma)}{n-k_n\sigma}\delta_{x_i^*},\sum_{i = 1}^n\frac{1}{n}\delta_{x_i}\right)\right],
    \]
    we have that
    \[
    \lambda_2 \leq \frac{n}{n - k_n \sigma}\leq \frac{n}{n(1-\sigma)}=\frac{1}{1-\sigma},
    \]
    and Wasserstein distance does not depend on $U_n.$ Note also that $\sum_{i = 1}^n \frac{1}{n}\delta_{x_i} = \sum_{i = 1}^{k_n}\frac{n_i}{n}\delta_{x_i^*}$. Since we can decompose
    \[
    \sum_{i = 1}^{k_n}\frac{n_i}{n}\delta_{x_i^*} = \sum_{i = 1}^{k_n} \frac{n_i-\sigma}{n}\delta_{x_i^*} + \sum_{i = 1}^{k_n} \frac{\sigma}{n}\delta_{x_i^*} = 
    \frac{n-k_n\sigma}{n}\sum_{i = 1}^{k_n} \frac{n_i - \sigma}{n - k_n\sigma}\delta_{x_i^*} + \frac{k_n \sigma}{n}\sum_{i = 1}^{k_n}\frac{1}{k_n} \delta_{x_i^*},
    \]
    using the convexity of the Wasserstein distance we get
    \begin{align*}
        \wtr\Bigg( \sum_{i = 1}^{k_n} \frac{(n_i-\sigma)}{n-k_n\sigma}&\delta_{x_i^*},\sum_{i = 1}^{k_n}\frac{n_i}{n}\delta_{x_i^*}\Bigg) = \\
        & = \wtr \Bigg( \frac{n-k_n\sigma}{n}\sum_{i = 1}^{k_n} \frac{n_i - \sigma}{n - k_n\sigma}\delta_{x_i^*} + \frac{k_n \sigma}{n}\sum_{i = 1}^{k_n}\frac{n_i - \sigma}{n - k_n\sigma} \delta_{x_i^*}, \frac{n-k_n\sigma}{n}\sum_{i = 1}^{k_n} \frac{n_i - \sigma}{n - k_n\sigma}\delta_{x_i^*} + \frac{k_n \sigma}{n}\sum_{i = 1}^{k_n}\frac{1}{k_n} \delta_{x_i^*}\Bigg) \\
        & = \frac{k_n \sigma}{n} \wtr\left( \sum_{i = 1}^{k_n} \frac{n_i - \sigma}{n - k_n\sigma}\delta_{x_i^*}, \sum_{i = 1}^{k_n}\frac{1}{k_n}\delta_{x_i^*}\right) \\
        & \leq 2 \frac{k_n\sigma}{n}.
    \end{align*}
    We conclude that $\E \mathcal{A}^U \mathbf{1}_{A_n} \lesssim \frac{k_n}{n}.$

    Now suppose that $U_n^\sigma < \alpha \frac{n - k_n \sigma}{n}.$ We decompose 
    \[
    \frac{\alpha}{\alpha + n}P_0 = \frac{ \sigma U_n^\sigma}{C_n}P_0 + \left( \frac{\alpha}{\alpha + n}- \frac{ \sigma U_n^\sigma}{C_n}\right) P_0.
    \]
    Define by 
    \[
    \lambda_1' = \frac{\alpha}{\alpha + n} - \frac{\sigma U_n^\sigma}{C_n}.
    \]
    Note that on $A_n^c,$ we have that $\lambda_1' \in (0,1).$ On $A_n^c,$
    \begin{align*}
        \wtr \Bigg( \frac{ \sigma U_n^\sigma}{C_n}P_0 & + \frac{n - k_n\sigma}{C_n}\sum_{i = 1}^{k_n}\frac{n_i - \sigma}{n - k_n\sigma}\delta_{x_i^*}, \\
        & \frac{ \sigma U_n^\sigma}{C_n}P_0 + \frac{n - k_n\sigma}{C_n}\left( \frac{C_n}{n - k_n\sigma}\lambda_1'P_0 + \frac{C_n}{n-k_n\sigma}\frac{n}{\alpha + n}\sum_{i = 1}^{k_n}\frac{n_i}{n}\delta_{x_i^*}\right) \Bigg) \\
        & \leq \frac{n - k_n\sigma}{C_n}\wtr \left( \sum_{i = 1}^{k_n}\frac{n_i - \sigma}{n - k_n\sigma}\delta_{x_i^*}, 
        \frac{C_n}{n - k_n\sigma}\lambda_1'P_0 + \frac{C_n}{n - k_n\sigma}\frac{n}{\alpha + n}\sum_{i = 1}^{k_n}\frac{n_i}{n}\delta_{x_i^*}\right) \\
        & \leq \lambda_1'\wtr\left( \sum_{i = 1}^{k_n}\frac{n_i - \sigma}{n - k_n\sigma}\delta_{x_i^*}, P_0\right) + \frac{n}{\alpha + n}\wtr \left(\sum_{i = 1}^{k_n}\frac{n_i - \sigma}{n - k_n \sigma}\delta_{x_i^*}, \sum_{i = 1}^{k_n}\frac{n_i}{n}\delta_{x_i^*} \right).
    \end{align*}
    As we can bound $\lambda_1'$ by $\frac{\alpha}{\alpha + n}$, using same reasoning as above, we deduce that $\E \mathcal{A}^U \mathbf{1}_{A_n} \lesssim \frac{k_n}{n}.$
\end{proof}
